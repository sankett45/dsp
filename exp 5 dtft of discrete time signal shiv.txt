EXPERIMENT-5A
 clear all; 
close all; 
clc; 
n=-50:50; 
%unit impulse 
for ii=1:length(n) 
    if n(ii)==0 
        x(ii)=1; 
    else 
        x(ii)=0; 
    end 
end 
 
%unit step 
for ii=1:length(n) 
    if n(ii)>=0 
        x(ii)=1; 
    else 
        x(ii)=0; 
    end 
end 
 
%exponential 
for ii=1:length(n) 
    if n(ii)>=0 
        x(ii)=exp(-0.2*n(ii)); 
    else 
        x(ii)=0; 
    end 
end 
%sinusoidal 
amp=1;M=1;N=10; 
for ii=1:length(n)    
x(ii)=amp*sin(2*pi*M/N*n(ii)); 
end 
%Gate 
N=5; 
for ii=1:length(n) 
    if(n(ii)<-N) 
        x(ii)=0; 
    elseif(n(ii)>N) 
        x(ii)=0; 
    else 
        x(ii)=1 
    end 
end 
 
%sinc 
N=5; 
for ii=1:length(n) 
    if(n(ii)==0) 
        x(ii)=1/N; 
    else 
        x(ii) = (1/N) * (sin(n(ii) * pi / 
N) / (n(ii) * pi / N)); 
    end 
end 
subplot(3,2,1:2) 
stem(n,x); 
xlabel("n"); 
ylabel("x[n]"); 
title("Discrete time signal"); 
omega=-4*pi:0.01:4*pi; 
xtemp=[]; 
X=[]; 
for a=1:length(omega); 
 
    for b=1:length(n); 
        xtemp = [xtemp x(b)*exp(
j*omega(a)*n(b))]; 
    end 
    xtemp=sum(xtemp); 
    X=[X xtemp]; 
    xtemp=[]; 
end 
magx=abs(X); 
subplot(3,2,3:4); 
plot(omega/pi,magx); grid; 
xlabel("frequence"); 
ylabel("x(omega)"); 
title("magnitude plot"); 
angx=angle(X); 
subplot(3,2,5:6); 
plot(omega/pi,angx); grid; 
xlabel("frequence in pi"); 
ylabel("angle(x)");
